  <script>
    // Nome do arquivo PDF — se você mudar o nome do arquivo, atualize aqui também:
    const PDF_FILE = "DocumentoCartilhaCuidarmais-combinado.pdf";

    const topicButtons = document.querySelectorAll(".topic-btn");
    const pdfViewer = document.getElementById("pdfViewer");
    const topicSearch = document.getElementById("topicSearch");

    function updatePdf(pageNumber) {
      // O parâmetro #page=n é interpretado pela maioria dos leitores de PDF do navegador
      pdfViewer.src = PDF_FILE + "#page=" + pageNumber + "&view=FitH";
    }

    topicButtons.forEach((btn) => {
      btn.addEventListener("click", () => {
        const page = btn.dataset.page;

        // Atualiza destaque visual
        topicButtons.forEach((b) => b.classList.remove("is-active"));
        btn.classList.add("is-active");

        // Atualiza o PDF
        updatePdf(page);
      });
    });

    // Filtro de tópicos pelo campo de busca
    if (topicSearch) {
      topicSearch.addEventListener("input", () => {
        const query = topicSearch.value.trim().toLowerCase();

        topicButtons.forEach((btn) => {
          const label = btn.textContent.toLowerCase();
          const item = btn.closest(".topic-item");
          if (!item) return;

          item.style.display = label.includes(query) ? "" : "none";
        });
      });
    }
  </script>
